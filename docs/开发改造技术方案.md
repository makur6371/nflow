## 一、项目概述

### 1.1 项目定位

nflow 是一个基于阿里 iflow-cli 架构深度改造的 AI 小说创作 CLI 工具。

**核心价值**：
- **靠谱**：三重自审查机制，解决长篇写作幻觉问题
- **免费**：集成 iflow 免费模型，零成本使用
- **强大**：200+ 专用生成器，100+ 写作模板

### 1.2 技术选型

| 类别 | 技术 | 版本 |
|------|------|------|
| 开发语言 | TypeScript | 5.0+ |
| 运行时 | Node.js | 20+ |
| UI 框架 | React via Ink | - |
| 构建工具 | esbuild | - |
| 包管理 | npm (workspaces) | - |
| 测试框架 | Vitest | - |
| 数据库 | SQLite | - |

### 5.3 字数控制系统

```typescript
class WordCountController {
  // 字数配置（基于番茄小说平台要求）
  config = {
    target: 2200,        // 目标字数
    min: 2000,           // 最小字数
    max: 2400,           // 最大字数（自动修复阈值）
    absoluteMax: 3000    // 绝对最大字数
  };

  // 实时统计字数
  countWords(text: string): number {
    return text.replace(/\s+/g, '').length;
  }

  // 检查字数是否达标
  checkWordCount(text: string): WordCountStatus {
    const count = this.countWords(text);
    if (count < this.config.min) {
      return { 
        status: 'insufficient', 
        count, 
        shortage: this.config.min - count 
      };
    } else if (count > this.config.max) {
      return { 
        status: 'excess', 
        count, 
        excess: count - this.config.max 
      };
    } else {
      return { status: 'optimal', count };
    }
  }

  // 自动扩展字数
  async expandContent(chapter: string, target: number): Promise<string> {
    const current = this.countWords(chapter);
    const needed = target - current;

    // 使用AI扩展内容
    const prompt = `
当前章节字数：${current}字
目标字数：${target}字
需要增加：${needed}字

请在保持原有内容和风格的基础上，自然地扩展以下章节，重点增加：
- 环境描写细节
- 人物心理活动
- 动作描写
- 感官体验

章节：
${chapter}
`;

    const expanded = await this.callAI(prompt);
    return expanded;
  }

  // 自动压缩字数
  async compressContent(chapter: string, target: number): Promise<string> {
    const current = this.countWords(chapter);
    const excess = current - target;

    // 使用AI压缩内容
    const prompt = `
当前章节字数：${current}字
目标字数：${target}字
需要删除：${excess}字

请在保持核心情节和风格的基础上，精简以下章节，重点：
- 删除冗余描写
- 合并相似内容
- 删除重复信息
- 精简对话

章节：
${chapter}
`;

    const compressed = await this.callAI(prompt);
    return compressed;
  }
}
```

### 5.4 质量把控闭环流程

```typescript
class QualityControlLoop {
  // 闭环流程
  async execute(chapter: string): Promise<QualityControlResult> {
    let currentChapter = chapter;
    const loopLog: LoopLog[] = [];
    let iteration = 0;
    const maxIterations = 3;

    while (iteration < maxIterations) {
      iteration++;
      
      // 步骤1：质量审核
      const review = await this.reviewChapter(currentChapter);
      loopLog.push({
        step: 'review',
        iteration,
        result: review
      });

      // 如果质量达标，退出循环
      if (review.score >= 80) {
        break;
      }

      // 步骤2：设定一致性检查
      const consistency = await this.checkConsistency(currentChapter);
      loopLog.push({
        step: 'consistency',
        iteration,
        result: consistency
      });

      // 步骤3：自动修复
      if (consistency.hasIssues) {
        const fixed = await this.autoFix(currentChapter, consistency.issues);
        currentChapter = fixed.content;
        loopLog.push({
          step: 'auto_fix',
          iteration,
          result: fixed
        });
      }

      // 步骤4：润色
      const polished = await this.polishChapter(currentChapter);
      currentChapter = polished.content;
      loopLog.push({
        step: 'polish',
        iteration,
        result: polished
      });
    }

    // 最终质量评分
    const finalScore = await this.reviewChapter(currentChapter);

    return {
      content: currentChapter,
      score: finalScore.score,
      iterations: iteration,
      log: loopLog,
      passed: finalScore.score >= 80
    };
  }

  // 质量审核
  async reviewChapter(chapter: string): Promise<ReviewResult> {
    const checks = await Promise.all([
      this.checkCharacterConsistency(chapter),
      this.checkPlotLogic(chapter),
      this.checkWorldSetting(chapter),
      this.checkTimeline(chapter),
      this.checkForeshadowing(chapter)
    ]);

    const scores = {
      characterConsistency: checks[0].score * 0.30,
      plotLogic: checks[1].score * 0.25,
      worldSetting: checks[2].score * 0.20,
      timeline: checks[3].score * 0.15,
      foreshadowing: checks[4].score * 0.10
    };

    const totalScore = Math.round(
      scores.characterConsistency +
      scores.plotLogic +
      scores.worldSetting +
      scores.timeline +
      scores.foreshadowing
    );

    return {
      score: totalScore,
      details: scores,
      issues: this.identifyIssues(checks)
    };
  }

  // 设定一致性检查
  async checkConsistency(chapter: string): Promise<ConsistencyResult> {
    const context = await this.getFullContext();
    const issues: Issue[] = [];

    // 检查人物一致性
    const characterIssues = await this.checkCharacters(chapter, context.characters);
    issues.push(...characterIssues);

    // 检查时间线
    const timelineIssues = await this.checkTimelineConsistency(chapter, context.timeline);
    issues.push(...timelineIssues);

    // 检查世界观设定
    const settingIssues = await this.checkWorldSettings(chapter, context.worldSettings);
    issues.push(...settingIssues);

    return {
      hasIssues: issues.length > 0,
      issues: issues
    };
  }

  // 自动修复
  async autoFix(chapter: string, issues: Issue[]): Promise<AutoFixResult> {
    let fixedChapter = chapter;
    const fixLog: FixLog[] = [];

    for (const issue of issues) {
      try {
        const fix = await this.fixIssue(fixedChapter, issue);
        fixedChapter = fix.content;
        fixLog.push({
          type: issue.type,
          success: true,
          action: fix.action
        });
      } catch (error) {
        fixLog.push({
          type: issue.type,
          success: false,
          action: 'failed',
          error: error.message
        });
      }
    }

    return {
      content: fixedChapter,
      log: fixLog
    };
  }

  // 润色
  async polishChapter(chapter: string): Promise<PolishResult> {
    const prompt = `
```

```typescript
class AutoFixer {
  // 自动修复入口
  async autoFix(chapter: string, issues: Issue[]): Promise<FixResult> {
    let fixedChapter = chapter;
    const fixLog: FixLog[] = [];

    for (const issue of issues) {
      try {
        const fix = await this.fixIssue(fixedChapter, issue);
        fixedChapter = fix.content;
        fixLog.push({
          type: issue.type,
          description: issue.description,
          action: fix.action,
          success: true
        });
      } catch (error) {
        fixLog.push({
          type: issue.type,
          description: issue.description,
          action: 'failed',
          success: false,
          error: error.message
        });
      }
    }

    return {
      content: fixedChapter,
      log: fixLog,
      success: fixLog.every(log => log.success)
    };
  }

  // 修复单个问题
  async fixIssue(chapter: string, issue: Issue): Promise<Fix> {
    switch (issue.type) {
      case 'word_count_insufficient':
        return this.fixWordCount(chapter, issue);
      case 'word_count_excess':
        return this.fixWordCountExcess(chapter, issue);
      case 'character_ooc':
        return this.fixCharacterOOC(chapter, issue);
      case 'timeline_error':
        return this.fixTimeline(chapter, issue);
      case 'setting_conflict':
        return this.fixSettingConflict(chapter, issue);
      case 'foreshadowing_missed':
        return this.fixForeshadowing(chapter, issue);
      default:
        throw new Error(`Unknown issue type: ${issue.type}`);
    }
  }

  // 修复字数不足
  async fixWordCount(chapter: string, issue: Issue): Promise<Fix> {
    const target = issue.targetWordCount || 2200;
    const controller = new WordCountController();
    const expanded = await controller.expandContent(chapter, target);

    return {
      content: expanded,
      action: `字数从${chapter.length}扩展到${expanded.length}`
    };
  }

  // 修复字数超标
  async fixWordCountExcess(chapter: string, issue: Issue): Promise<Fix> {
    const target = issue.targetWordCount || 2200;
    const controller = new WordCountController();
    const compressed = await controller.compressContent(chapter, target);

    return {
      content: compressed,
      action: `字数从${chapter.length}压缩到${compressed.length}`
    };
  }

  // 修复人物OOC
  async fixCharacterOOC(chapter: string, issue: Issue): Promise<Fix> {
    const character = issue.character;
    const profile = await this.getCharacterProfile(character);

    const prompt = `
当前章节中，${character}的表现与设定不符。

人物设定：
- 性格：${profile.traits}
- 背景：${profile.background}
- 语言风格：${profile.speechStyle}

问题段落：
${issue.problemParagraph}

请修改这段内容，使人物表现符合设定。
`;

    const fixed = await this.callAI(prompt);
    return {
      content: chapter.replace(issue.problemParagraph, fixed),
      action: `修正${character}的行为表现`
    };
  }

  // 修复时间线错误
  async fixTimeline(chapter: string, issue: Issue): Promise<Fix> {
    const timeline = await this.getTimeline();
    const prompt = `
当前章节中存在时间线错误。

正确的时间线：
${timeline.map(event => `${event.time}: ${event.description}`).join('\n')}

错误描述：
${issue.description}

请修正时间线错误。
`;

    const fixed = await this.callAI(prompt);
    return {
      content: fixed,
      action: '修正时间线错误'
    };
  }

  // 修复设定冲突
  async fixSettingConflict(chapter: string, issue: Issue): Promise<Fix> {
    const settings = await this.getWorldSettings();
    const conflictSetting = settings.find(s => s.key === issue.conflictKey);

    const prompt = `
当前章节中存在设定冲突。

正确设定：
- ${issue.conflictKey}: ${conflictSetting.value}

冲突描述：
${issue.description}

请修正设定冲突。
`;

    const fixed = await this.callAI(prompt);
    return {
      content: fixed,
      action: '修正设定冲突'
    };
  }

  // 修复伏笔遗漏
  async fixForeshadowing(chapter: string, issue: Issue): Promise<Fix> {
    const foreshadowing = await this.getForeshadowing(issue.foreshadowingId);

当前章节需要呼应伏笔。

伏笔信息：
- ID: ${foreshadowing.id}
- 描述：${foreshadowing.description}
- 原章节：${foreshadowing.chapter}

请在本章中自然地呼应这个伏笔。
`;
```
